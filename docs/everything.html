    Off-grid Mechanicals Vault

    Every camp-type structure would benefit from a temporary and small electricity supply and running hot water. A minimalistic vault design can help simplify the setup, maintenance, and constructability of this. Obviously, every design parameter depends on conservation and reduction of need. A dry temp-stable space is also a great spot for emergency supplies.
    1. Structure:
      1. Constructability: 4 ft cube built of double-layer waterproof plywood, with spray foam in between
      2. 95% buried for thermal efficiency, low noise, and security
      3. Accessible and visible top (so no one runs it over) yet protected from the elements and vandalism. Could also site structure on top of it, and size it for lock-box usage as well.
      4. insulated and heated for consistent temp and low humidity
    2. Mechanicals
      1. Battery bank: lithium
      2. Well: some people do this but a lot depends on siting for soil type and depth to water table
      3. Pump: 12v, need not be large capacity
      4. Pressure Tank
        1. this shouldn't be large, as a portable water tower should be used for increased capacity
      5. Filter: 50 micron fiber followed by 5 micron charcoal to remove sediment, chemicals, and protozoa
      6. UV: to remove bacteria and viruses
      7. Propane tankless water heater:
        1. if its a long pipe run, may want to move inside
        2. solar preheat coils?
      8. Power:
        1. 12v for safety and convenience
        2. charger that can handle solar, genset, alternator, or wind turbine inputs
    3. Monitoring:
      1. power level and charge rates
      2. filter pressure drop
      3. UV beam strength
      4. vault temp and humidity
      5. telemetery: wifi? feather LoRa?
      6. pressure tank back pressure
      7. vault door alarm
      8. propane and CO monitors
      9. well and dry-well levels and temp
      10. pump draw, flow rate, and pressure
      11. propane tank pressure

def parsePath(path):
    _from, _to = path.strip().split(" > ")
    _fromPort = _from.split(".")[-1]
    _fromAdd = _from[:-1*(len(_fromPort)+1)]
    _fromContext = domainLookup(_fromAdd)
    _toPort = _to.split(".")[-1]
    _toAdd = _to[:-1*(len(_toPort)+1)]
    _toContext = domainLookup(_toAdd)

void loop() {
  int8_t jobDuration;
  Adafruit_MQTT_Subscribe *subscription;

  // ping adafruit io a few times to make sure we remain connected
  if(! mqtt.ping(3)) {
    // reconnect to adafruit io
    if(! mqtt.connected())
      MQTT_connect();
  }
  //
  
  // this is our 'wait for incoming subscription packets' busy subloop

  while (subscription = mqtt.readSubscription(1000)) {

    if (subscription == &onoffbutton) {
      Serial.print(F("On-Off button: "));
      Serial.println((char *)onoffbutton.lastread);

      // convert mqtt ascii payload to int
      char *value = (char *)onoffbutton.lastread;
      Serial.print(F("Received: "));
      Serial.println(value);

      // Apply message to onoffbutton
      String message = String(value);
      message.trim();
      if (message == "ON") {
        jobDuration= Flood();      
        Serial.println("publishing to /get");
        onoff_get.publish((int32_t)jobDuration);
        break;
      }
    }
  }
}

void MQTT_connect() {
  int8_t ret;

  while ((ret = mqtt.connect()) != 0) {

    switch (ret) {
      case 1: Serial.println(F("Wrong protocol")); break;
      case 2: Serial.println(F("ID rejected")); break;
      case 3: Serial.println(F("Server unavail")); break;
      case 4: Serial.println(F("Bad user/pass")); break;
      case 5: Serial.println(F("Not authed")); break;
      case 6: Serial.println(F("Failed to subscribe")); break;
      default: Serial.println(F("Connection failed")); break;
    }

    if(ret >= 0)
      mqtt.disconnect();

    Serial.println(F("Retrying MQTT connection in 5 seconds..."));
    delay(5000);

  }
  Serial.println(F("MQTT Connected to Adafruit IO!"));

  /*
  Publishing to {username}/feeds/onoff/get will trigger the last
  value to be resent. Putting this call inside the MQTT_connect
  function means that whenever MQTT reconnects, the latest value
  for the feed will be received almost immediately.
  */
  Serial.println("publishing to /get");
  onoff_get.publish((int32_t)0);
}
